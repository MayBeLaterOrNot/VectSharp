208  /// <summary>
209  /// <para>Signals to this object that its shape has changed a new path needs to be computed for the purpose of hit-testing.
210  /// Also signals to the <see cref="Parent"/> that the visual properties of this object have changed and triggers a redraw.</para>
211  /// <para>This method should be called whenever the "shape" of the object represented by the <see cref="SKRenderAction"/> changes.
212  /// If only the visual properties of this object have changed (e.g. the colour), call the <see cref="InvalidateVisual"/> method instead.</para>
213  /// <para>If you make changes to more than one <see cref="SKRenderAction"/> contained in the same <see cref="Canvas"/>, you only need to invalidate the last one.</para>
214  /// <para>This method should only be called after the output has been fully initialized.</para>

220  this.InvalidateVisual();

223  internal void CreateHitTestPath()
224  {
225  if (this.ActionType == ActionTypes.Path)
226  {
227  if (this.Path != null && this.Paint != null)
228  {
229  this.HitTestPath?.Dispose();
230  this.HitTestPath = this.Paint.GetFillPath(this.Path);
231  }
232  }
233  else if (this.ActionType == ActionTypes.RasterImage)
234  {
235  if (this.ImageDestination != null)
236  {
237  this.HitTestPath?.Dispose();
238  this.HitTestPath = new SKPath();
239  this.HitTestPath.AddRect(this.ImageDestination.Value);
240  }
241  }
242  else if (this.ActionType == ActionTypes.Text)
243  {
244  if (this.Paint != null && this.Font != null)
245  {
246  this.HitTestPath?.Dispose();
247  this.Paint.Typeface = this.Font.Typeface;
248  this.Paint.TextSize = this.Font.Size;
249 
250  using (SKPath pth = this.Paint.GetTextPath(this.Text, this.TextX, this.TextY))
251  {
252  this.HitTestPath = this.Paint.GetFillPath(pth);
253  }
254  }
255  }

632  public void StrokeSimpleText(string text, double x, double y)
633  {
634  Utils.CoerceNaNAndInfinityToZero(ref x, ref y);
635 
636  if ((_textOption == AvaloniaContextInterpreter.TextOptions.NeverConvert || (_textOption == AvaloniaContextInterpreter.TextOptions.ConvertIfNecessary && (Font.FontFamily.IsStandardFamily || Font.FontFamily.TrueTypeFile?.FontStream != null))) && !System.Runtime.InteropServices.RuntimeInformation.IsOSPlatform(System.Runtime.InteropServices.OSPlatform.Linux))
637  {
638  SKTypeface typeface = SKTypefaceCache.GetSKTypeface(Font.FontFamily);
639 
640  double top = y;
641  double left = x;
642 
643  Font.DetailedFontMetrics metrics = Font.MeasureTextAdvanced(text);
644 
645  if (TextBaseline == TextBaselines.Top)
646  {
647  if (Font.FontFamily.TrueTypeFile != null)
648  {
649  left -= metrics.LeftSideBearing;
650  top += metrics.Top;
651  }
652  }
653  else if (TextBaseline == TextBaselines.Middle)
654  {
655  if (Font.FontFamily.TrueTypeFile != null)
656  {
657  left -= metrics.LeftSideBearing;
658  top += (metrics.Top + metrics.Bottom) * 0.5;
659  }
660  }
661  else if (TextBaseline == TextBaselines.Baseline)
662  {
663  if (Font.FontFamily.TrueTypeFile != null)
664  {
665  left -= metrics.LeftSideBearing;
666  }
667  }
668  else if (TextBaseline == TextBaselines.Bottom)
669  {
670  if (Font.FontFamily.TrueTypeFile != null)
671  {
672  left -= metrics.LeftSideBearing;
673  top += metrics.Bottom;
674  }
