149  private void Render()
150  {
151  if (Document != null)
152  {
153  double width = Math.Min(MaxRenderWidth, Math.Max(MinRenderWidth, this.Bounds.Width - MinVariation - 13));
154 
155  if (forcedRerender || double.IsNaN(lastRenderedWidth) || width != lastRenderedWidth && width < lastRenderedWidth - MinVariation || width > lastRenderedWidth + MinVariation)
156  {
157  Page pag = Renderer.RenderSinglePage(this.Document, width, out Dictionary<string, string> linkDestinations);
158 
159  Dictionary<string, Delegate> taggedActions = new Dictionary<string, Delegate>();
160  Dictionary<string, Avalonia.Point> linkDestinationPoints = new Dictionary<string, Avalonia.Point>();
161 
162  foreach (KeyValuePair<string, string> linkDestination in linkDestinations)
163  {
164  string url = linkDestination.Value;
165 
166  taggedActions.Add(linkDestination.Key, (Func<RenderAction, IEnumerable<RenderAction>>)(act =>
167  {
168  act.PointerEnter += (s, e) =>
169  {
170  act.Parent.Cursor = new Avalonia.Input.Cursor(Avalonia.Input.StandardCursorType.Hand);
171  };
172 
173  act.PointerLeave += (s, e) =>
174  {
175  act.Parent.Cursor = new Avalonia.Input.Cursor(Avalonia.Input.StandardCursorType.Arrow);
176  };
177 
178  act.PointerPressed += (s, e) =>
179  {
180  if (url.StartsWith("#"))
181  {
182  if (linkDestinationPoints.TryGetValue(url.Substring(1), out Avalonia.Point target))
183  {
184  ScrollViewer scrollViewer = this.FindControl<ScrollViewer>("ScrollViewer");
185 
186  scrollViewer.Offset = new Vector(Math.Max(Math.Min(scrollViewer.Offset.X, target.X), target.X - scrollViewer.Viewport.Width), target.Y);
187  }
188  }
189  else
190  {
191  System.Diagnostics.Process.Start(new System.Diagnostics.ProcessStartInfo() { FileName = url, UseShellExecute = true });
192  }
193  };
194 
195  if (act.ActionType == RenderAction.ActionTypes.Path)
196  {
197  linkDestinationPoints[linkDestination.Key] = act.Geometry.Bounds.TopLeft.Transform(act.Transform);
198  }
199  else if (act.ActionType == RenderAction.ActionTypes.Text)
200  {
201  linkDestinationPoints[linkDestination.Key] = act.Text.Bounds.TopLeft.Transform(act.Transform);
202  }
203  else if (act.ActionType == RenderAction.ActionTypes.RasterImage)
204  {
205  linkDestinationPoints[linkDestination.Key] = act.ImageDestination.Value.TopLeft.Transform(act.Transform);
206  }
207 
208  return new RenderAction[] { act };
209  }));
210 
211  if (url.StartsWith("#"))
212  {
213  if (!taggedActions.ContainsKey(url.Substring(1)))
214  {
215  taggedActions.Add(url.Substring(1), (Func<RenderAction, IEnumerable<RenderAction>>)(act =>
216  {
217  if (act.ActionType == RenderAction.ActionTypes.Path)
218  {
219  linkDestinationPoints[url.Substring(1)] = act.Geometry.Bounds.TopLeft.Transform(act.Transform);
220  }
221  else if (act.ActionType == RenderAction.ActionTypes.Text)
222  {
223  linkDestinationPoints[url.Substring(1)] = act.Text.Bounds.TopLeft.Transform(act.Transform);
224  }
225  else if (act.ActionType == RenderAction.ActionTypes.RasterImage)
226  {
227  linkDestinationPoints[url.Substring(1)] = act.ImageDestination.Value.TopLeft.Transform(act.Transform);
228  }
229 
230  return new RenderAction[] { act };
231  }));
232  }
233  }
234  }
235 
236  Avalonia.Controls.Canvas can = pag.PaintToCanvas(false, taggedActions, false);
237 
238  this.FindControl<ScrollViewer>("ScrollViewer").Content = can;
239  this.FindControl<ScrollViewer>("ScrollViewer").Padding = new Thickness(0, 0, 0, 0);
240  lastRenderedWidth = width;
241  forcedRerender = false;
242  }
243  else
244  {
245  this.FindControl<ScrollViewer>("ScrollViewer").Padding = new Thickness(0, 0, width - lastRenderedWidth, 0);
