89  {
90  double currDelta = 0;
91  double pathLength = path.MeasureLength();
92 
93  Font.DetailedFontMetrics fullMetrics = font.MeasureTextAdvanced(text);
94 
95  switch (anchor)
96  {
97  case TextAnchors.Left:
98  break;
99  case TextAnchors.Center:
100  currDelta = -fullMetrics.Width * 0.5 / pathLength;
101  break;
102  case TextAnchors.Right:
103  currDelta = -fullMetrics.Width / pathLength;
104  break;
105  }
106 
107  for (int i = 0; i < text.Length; i++)
108  {
109  string c = text.Substring(i, 1);
110 
111  Font.DetailedFontMetrics metrics = font.MeasureTextAdvanced(c);
112 
113  Point origin = path.GetPointAtRelative(reference + currDelta);
114 
115  Point tangent = path.GetTangentAtRelative(reference + currDelta + (metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing) / pathLength * 0.5);
116 
117  this.Save();
118 
119  this.Translate(origin);
120  this.Rotate(Math.Atan2(tangent.Y, tangent.X));
121 
122  switch (textBaseline)
123  {
124  case TextBaselines.Top:
125  if (i > 0)
126  {
127  this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Top), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
128  }
129  else
130  {
131  this.FillText(new Point(0, fullMetrics.Top), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
132  }
133  break;
134  case TextBaselines.Baseline:
135  if (i > 0)
136  {
137  this.FillText(new Point(metrics.LeftSideBearing, 0), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
138  }
139  else
140  {
141  this.FillText(new Point(0, 0), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
142  }
143  break;
144  case TextBaselines.Bottom:
145  if (i > 0)
146  {
147  this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Bottom), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
148  }
149  else
150  {
151  this.FillText(new Point(0, fullMetrics.Bottom), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
152  }
153  break;
154  case TextBaselines.Middle:
155  if (i > 0)
156  {
157  this.FillText(new Point(metrics.LeftSideBearing, fullMetrics.Bottom + fullMetrics.Height / 2), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
158  }
159  else
160  {
161  this.FillText(new Point(0, fullMetrics.Bottom + fullMetrics.Height / 2), c, font, fillColour, textBaseline: TextBaselines.Baseline, tag);
162  }
163  break;
164  }
165 
166  this.Restore();
167 
168  if (i > 0)
169  {
170  currDelta += (metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing) / pathLength;
171  }
172  else
173  {
174  currDelta += (metrics.Width + metrics.RightSideBearing) / pathLength;
175  }
176  }

285  /// <summary>
286  /// Fill a formatted text string.
287  /// </summary>
288  /// <param name="origin">The text origin. See <paramref name="textBaseline"/>.</param>
289  /// <param name="text">The <see cref="FormattedText"/> to draw.</param>
290  /// <param name="fillColour">The default <see cref="Brush"/> to use to fill the text. This can be overridden by each <paramref name="text"/> element.</param>
291  /// <param name="textBaseline">The text baseline (determines what the vertical component of <paramref name="origin"/> represents).</param>
292  /// <param name="tag">A tag to identify the filled text.</param>
293  public void FillText(Point origin, IEnumerable<FormattedText> text, Brush fillColour, TextBaselines textBaseline = TextBaselines.Top, string tag = null)
294  {
295  List<FormattedText> enumeratedText = new List<FormattedText>();
296  List<Font.DetailedFontMetrics> allMetrics = new List<Font.DetailedFontMetrics>();
297 
298  Font.DetailedFontMetrics fullMetrics = text.Measure(enumeratedText, allMetrics);
299 
300  Point baselineOrigin = origin;
301 
302  switch (textBaseline)
303  {
304  case TextBaselines.Baseline:
305  baselineOrigin = origin;
306  break;
307  case TextBaselines.Top:
308  baselineOrigin = new Point(origin.X, origin.Y + fullMetrics.Top);
309  break;
310  case TextBaselines.Bottom:
311  baselineOrigin = new Point(origin.X, origin.Y + fullMetrics.Bottom);
312  break;
313  case TextBaselines.Middle:
314  baselineOrigin = new Point(origin.X, origin.Y + fullMetrics.Top * 0.5 + fullMetrics.Bottom * 0.5);
315  break;
316  }
317 
318  for (int i = 0; i < enumeratedText.Count; i++)
319  {
320  FormattedText txt = enumeratedText[i];
321 
322  if (txt.Script == Script.Normal)
323  {
324  Font.DetailedFontMetrics metrics = allMetrics[i];
325 
326  if (i > 0)
327  {
328  FillText(baselineOrigin.X + metrics.LeftSideBearing, baselineOrigin.Y, txt.Text, txt.Font, txt.Brush ?? fillColour, TextBaselines.Baseline, tag);
329  }
330  else
331  {
332  FillText(baselineOrigin, txt.Text, txt.Font, txt.Brush ?? fillColour, TextBaselines.Baseline, tag);
333  }
334 
335  if (i > 0)
336  {
337  baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing, baselineOrigin.Y);
338  }
339  else
340  {
341  baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing, baselineOrigin.Y);
342  }
343  }
344  else
345  {
346  Font newFont = new Font(txt.Font.FontFamily, txt.Font.FontSize * 0.7);
347 
348  Font.DetailedFontMetrics metrics = allMetrics[i];
349 
350  if (i == 0)
351  {
352  baselineOrigin = new Point(baselineOrigin.X - metrics.LeftSideBearing, baselineOrigin.Y);
353  }
354 
355  if (txt.Script == Script.Subscript)
356  {
357  FillText(baselineOrigin.X + metrics.LeftSideBearing, baselineOrigin.Y + txt.Font.FontSize * 0.14, txt.Text, newFont, txt.Brush ?? fillColour, TextBaselines.Baseline, tag);
358  }
359  else if (txt.Script == Script.Superscript)
360  {
361  FillText(baselineOrigin.X + metrics.LeftSideBearing, baselineOrigin.Y - txt.Font.FontSize * 0.33, txt.Text, newFont, txt.Brush ?? fillColour, TextBaselines.Baseline, tag);
362  }
363 
364 
365  if (i > 0)
366  {
367  baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing + metrics.LeftSideBearing, baselineOrigin.Y);
368  }
369  else
370  {
371  baselineOrigin = new Point(baselineOrigin.X + metrics.Width + metrics.RightSideBearing, baselineOrigin.Y);
372  }
373  }
374  }
375  }
