359  /// <summary>
360  /// Convert a CSS colour string into a <see cref="Colour"/> object.
361  /// </summary>
362  /// <param name="cssString">The CSS colour string. In addition to 148 standard colour names (case-insensitive), #RGB, #RGBA, #RRGGBB and #RRGGBBAA hex strings and rgb(r, g, b) and rgba(r, g, b, a) functional colour notations are supported.</param>
363  /// <returns></returns>
364  public static Colour? FromCSSString(string cssString)
365  {
366  if (cssString.StartsWith("#"))
367  {
368  cssString = cssString.Substring(1);
369 
370  if (cssString.Length == 3)
371  {
372  byte r = byte.Parse(cssString.Substring(0, 1) + cssString.Substring(0, 1), System.Globalization.NumberStyles.HexNumber);
373  byte g = byte.Parse(cssString.Substring(1, 1) + cssString.Substring(1, 1), System.Globalization.NumberStyles.HexNumber);
374  byte b = byte.Parse(cssString.Substring(2, 1) + cssString.Substring(2, 1), System.Globalization.NumberStyles.HexNumber);
375 
376  return Colour.FromRgb(r, g, b);
377  }
378  else if (cssString.Length == 4)
379  {
380  byte r = byte.Parse(cssString.Substring(0, 1) + cssString.Substring(0, 1), System.Globalization.NumberStyles.HexNumber);
381  byte g = byte.Parse(cssString.Substring(1, 1) + cssString.Substring(1, 1), System.Globalization.NumberStyles.HexNumber);
382  byte b = byte.Parse(cssString.Substring(2, 1) + cssString.Substring(2, 1), System.Globalization.NumberStyles.HexNumber);
383  byte a = byte.Parse(cssString.Substring(3, 1) + cssString.Substring(3, 1), System.Globalization.NumberStyles.HexNumber);
384 
385  return Colour.FromRgba(r, g, b, a);
386  }
387  else if (cssString.Length == 6)
388  {
389  byte r = byte.Parse(cssString.Substring(0, 2), System.Globalization.NumberStyles.HexNumber);
390  byte g = byte.Parse(cssString.Substring(2, 2), System.Globalization.NumberStyles.HexNumber);
391  byte b = byte.Parse(cssString.Substring(4, 2), System.Globalization.NumberStyles.HexNumber);
392 
393  return Colour.FromRgb(r, g, b);
394  }
395  else if (cssString.Length == 8)
396  {
397  byte r = byte.Parse(cssString.Substring(0, 2), System.Globalization.NumberStyles.HexNumber);
398  byte g = byte.Parse(cssString.Substring(2, 2), System.Globalization.NumberStyles.HexNumber);
399  byte b = byte.Parse(cssString.Substring(4, 2), System.Globalization.NumberStyles.HexNumber);
400  byte a = byte.Parse(cssString.Substring(6, 2), System.Globalization.NumberStyles.HexNumber);
401 
402  return Colour.FromRgba(r, g, b, a);
403  }
404  else
405  {
406  return null;
407  }
408  }
409  else if (cssString.StartsWith("rgb(") || cssString.StartsWith("rgba("))
410  {
411  try
412  {
413  cssString = cssString.Substring(cssString.IndexOf("(") + 1).Replace(")", "").Replace(" ", "");
414  string[] splitCssString = cssString.Split(',');
415 
416  double R = ParseColourValueOrPercentage(splitCssString[0]);
417  double G = ParseColourValueOrPercentage(splitCssString[1]);
418  double B = ParseColourValueOrPercentage(splitCssString[2]);
419 
420  double A = 1;
421 
422  if (splitCssString.Length == 4)
423  {
424  A = double.Parse(splitCssString[3], System.Globalization.CultureInfo.InvariantCulture);
425  }
426 
427  return Colour.FromRgba(R, G, B, A);
428  }
429  catch
430  {
431  return null;
432  }
433  }
434  else
435  {
436  if (StandardColours.TryGetValue(cssString, out Colour tbr))
437  {
438  return tbr;
439  }
440  else
441  {
442  return null;
443  }
444  }
445  }
