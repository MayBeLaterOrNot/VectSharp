122  /// Creates a sphere.
123  /// </summary>
124  /// <param name="center">The centre of the sphere.</param>
125  /// <param name="radius">The radius of the sphere.</param>
126  /// <param name="steps">The number of meridians and parallels to use when generating the sphere.</param>
127  /// <param name="fill">A collection of materials that will be applied to the <see cref="Triangle3DElement"/>s returned by this method.</param>
128  /// <param name="tag">A tag that will be applied to the <see cref="Triangle3DElement"/>s returned by this method.</param>
129  /// <param name="zIndex">A z-index that will be applied to the <see cref="Triangle3DElement"/>s returned by this method.</param>
130  /// <returns>A list of <see cref="Triangle3DElement"/>s that constitute the sphere.</returns>
131  public static List<Element3D> CreateSphere(Point3D center, double radius, int steps, IEnumerable<IMaterial> fill, string tag = null, int zIndex = 0)
132  {
133  List<Point3D> points = new List<Point3D>();
134 
135  for (int t = 0; t <= steps; t++)
136  {
137  for (int p = 0; p < steps * 2; p++)
138  {
139  double theta = Math.PI / steps * t;
140  double phi = Math.PI / steps * p;
141 
142  double x = center.X + radius * Math.Sin(theta) * Math.Cos(phi);
143  double y = center.Y + radius * Math.Sin(theta) * Math.Sin(phi);
144  double z = center.Z + radius * Math.Cos(theta);
145 
146  points.Add(new Point3D(x, y, z));
147 
148  if (t == 0 || t == steps)
149  {
150  break;
151  }
152  }
153  }
154 
155  List<Element3D> tbr = new List<Element3D>(4 * steps + (points.Count - 2 - 2 * steps) * 2);
156 
157  for (int i = 0; i < points.Count - 1; i++)
158  {
159  if (i == 0)
160  {
161  for (int j = 0; j < 2 * steps; j++)
162  {
163  Point3D p1 = points[i];
164  Point3D p3 = points[i + 1 + j];
165  Point3D p2 = points[i + 1 + (j + 1) % (2 * steps)];
166 
167  Triangle3DElement tri = new Triangle3DElement(p1, p2, p3, (center - p1).Normalize(), (center - p2).Normalize(), (center - p3).Normalize());
168  tri.Fill.AddRange(fill);
169  tri.Tag = tag;
170  tri.ZIndex = zIndex;
171  tbr.Add(tri);
172  }
173  }
174  else if (i >= points.Count - 1 - 2 * steps)
175  {
176  Point3D p1 = points[i];
177  Point3D p3 = points[points.Count - 1];
178  Point3D p2 = points[points.Count - 1 - 2 * steps + (i - (points.Count - 1 - 2 * steps) + 1) % (2 * steps)];
179 
180  Triangle3DElement tri = new Triangle3DElement(p1, p2, p3, (center - p1).Normalize(), (center - p2).Normalize(), (center - p3).Normalize());
181  tri.Fill.AddRange(fill);
182  tri.Tag = tag;
183  tri.ZIndex = zIndex;
184  tbr.Add(tri);
185  }
186  else
187  {
188  if ((i - 1) % (2 * steps) < 2 * steps - 1)
189  {
190  Point3D p4 = points[i + 2 * steps];
191  Point3D p3 = points[i + 2 * steps + 1];
192  Point3D p2 = points[i + 1];
193  Point3D p1 = points[i];
194 
195  tbr.AddRange(CreateRectangle(p1, p2, p3, p4, (center - p1).Normalize(), (center - p2).Normalize(), (center - p3).Normalize(), (center - p4).Normalize(), fill, tag, zIndex));
196  }
197  else
198  {
199  Point3D p4 = points[i + 2 * steps];
200  Point3D p3 = points[(i / (2 * steps)) * 2 * steps + 1];
201  Point3D p2 = points[(i / (2 * steps) - 1) * 2 * steps + 1];
202  Point3D p1 = points[i];
203 
204  tbr.AddRange(CreateRectangle(p1, p2, p3, p4, (center - p1).Normalize(), (center - p2).Normalize(), (center - p3).Normalize(), (center - p4).Normalize(), fill, tag, zIndex));
205  }
206  }
207  }
208 
209  return tbr;
210  }

284  /// <summary>

286  /// </summary>
287  /// <param name="polygonBase2D">A 2D <see cref="GraphicsPath"/> representing the base of the prism.</param>
288  /// <param name="triangulationResolution">The resolution that will be used to linearise curve segments in the <see cref="GraphicsPath"/>.</param>

294  /// <param name="tag">A tag that will be applied to the <see cref="Triangle3DElement"/>s returned by this method.</param>

297  public static List<Element3D> CreatePrism(GraphicsPath polygonBase2D, double triangulationResolution, Point3D bottomOrigin, Point3D topOrigin, NormalizedVector3D baseXAxis, NormalizedVector3D baseYAxis, IEnumerable<IMaterial> fill, string tag = null, int zIndex = 0)
298  {
299  baseXAxis = (baseXAxis - baseYAxis * (baseXAxis * baseYAxis)).Normalize();
300 
301  List<Element3D> tbr = new List<Element3D>();
302 
303  bool orientation = (baseXAxis ^ baseYAxis) * (bottomOrigin - topOrigin) > 0;
304 
305  double[,] matrix1 = Matrix3D.RotationToAlignAWithB(new NormalizedVector3D(0, 1, 0), baseYAxis);
306  double[,] matrix2 = Matrix3D.RotationToAlignAWithB(((Vector3D)(matrix1 * new Point3D(1, 0, 0))).Normalize(), baseXAxis);
307 
308  List<List<NormalizedVector3D>> normals = (from el2 in polygonBase2D.GetLinearisationPointsNormals(triangulationResolution) select (from el in el2 select (el.X * baseXAxis + el.Y * baseYAxis).Normalize()).ToList()).ToList();
309 
310  polygonBase2D = polygonBase2D.Linearise(triangulationResolution);
311 
312  tbr.AddRange(CreatePolygon(polygonBase2D, triangulationResolution, bottomOrigin, baseXAxis, baseYAxis, orientation, fill, tag, zIndex));
313  tbr.AddRange(CreatePolygon(polygonBase2D, triangulationResolution, topOrigin, baseXAxis, baseYAxis, !orientation, fill, tag, zIndex));
314 
315  List<List<Point3D>> bottomPoints = (from el2 in polygonBase2D.GetPoints() select (from el in el2 select (Point3D)(el.X * baseXAxis + el.Y * baseYAxis + (Vector3D)bottomOrigin)).ToList()).ToList();
316  List<List<Point3D>> topPoints = (from el2 in polygonBase2D.GetPoints() select (from el in el2 select (Point3D)(el.X * baseXAxis + el.Y * baseYAxis + (Vector3D)topOrigin)).ToList()).ToList();
317 
318  if (orientation)
319  {
320  for (int i = 0; i < bottomPoints.Count; i++)
321  {
322  for (int j = 0; j < bottomPoints[i].Count - 1; j++)
323  {
324  tbr.AddRange(CreateRectangle(bottomPoints[i][j], bottomPoints[i][j + 1], topPoints[i][j + 1], topPoints[i][j], normals[i][j], normals[i][j + 1], normals[i][j + 1], normals[i][j], fill, tag, zIndex));
325  }
326  }
327  }
328  else
329  {
330  for (int i = 0; i < bottomPoints.Count; i++)
331  {
332  for (int j = 0; j < bottomPoints[i].Count - 1; j++)
333  {
334  tbr.AddRange(CreateRectangle(bottomPoints[i][j], topPoints[i][j], topPoints[i][j + 1], bottomPoints[i][j + 1], normals[i][j], normals[i][j], normals[i][j + 1], normals[i][j + 1], fill, tag, zIndex));
335  }
336  }
337  }
338 
339  return tbr;
